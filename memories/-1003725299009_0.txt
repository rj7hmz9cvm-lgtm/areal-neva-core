Фрагмент кода ниже:

```python
from telegram.ext import Updater, CommandHandler, MessageHandler, Filters
from googleapiclient.discovery import build
from google.oauth2 import service_account
import os
import httplib2
import mimetypes

# Настройки
GOOGLE_DRIVE_FOLDER_ID = 'your_folder_id'
SCOPES = ['https://www.googleapis.com/auth/drive']
SERVICE_ACCOUNT_FILE = 'credentials.json'

# Google Drive API
credentials = service_account.Credentials.from_service_account_file(SERVICE_ACCOUNT_FILE, scopes=SCOPES)
service = build('drive', 'v3', credentials=credentials)

def upload_to_gdrive(file_path, file_name):
    """Загружает файл в Google Drive."""
    file_metadata = {'name': file_name, 'parents': [GOOGLE_DRIVE_FOLDER_ID]}
    media = googleapiclient.http.MediaFileUpload(file_path, mimetype=mimetypes.guess_type(file_path)[0])
    service.files().create(body=file_metadata, media_body=media, fields='id').execute()

def handle_file(update, context):
    """Сохраняет файл на компьютере и загружает в Google Drive."""
    file = update.message.document or update.message.photo[-1] or update.message.video or update.message.audio
    file_name = file.file_name if hasattr(file, 'file_name') else f"{file.file_id}.{mimetypes.guess_extension(file.mime_type)}"
    file_path = os.path.join(os.path.expanduser("~/Downloads"), file_name)
    file.download(file_path)
    upload_to_gdrive(file_path, file_name)
    os.remove(file_path)
    update.message.reply_text(f"Файл загружен в Google Drive: {file_name}")

def main():
    updater = Updater("YOUR_TELEGRAM_BOT_TOKEN")
    dp = updater.dispatcher
    dp.add_handler(MessageHandler(Filters.document | Filters.photo | Filters.video | Filters.audio, handle_file))
    updater.start_polling()
    updater.idle()

if __name__ == '__main__':
    main()
```
